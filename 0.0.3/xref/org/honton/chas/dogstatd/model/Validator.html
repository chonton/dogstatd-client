6    * Validation routines

8   final class Validator {

10    // Metric names must start with a letter, and after that may contain ascii alphanumerics,
11    // underscore and periods.
12    private static final Pattern VALID_NAME = Pattern.compile("^[A-Za-z][0-9A-Za-z_\\.]*$");
13  
14    // Tags must start with a lowercase letter, and after that may contain lowercase alphanumerics,
15    // underscores, minuses, colons, periods and slashes.
16    // Tags can be up to 200 characters long.
17    private static final Pattern VALID_TAG = Pattern.compile("^[a-z][0-9a-z_\\-:\\./]{0,199}$");
18  
19    // No pipes
20    private static final Pattern NO_PIPES = Pattern.compile("^[^\\|\\p{Cntrl}]+$");
21  
22    private Validator() {}
23  
24    /**
25     * Validate a metric name.
26     * 
27     * @param name The name to validate.
28     * @return true, if the name start with an ASCII letter, and contains only ASCII alphanumerics,
29     *         underscores, and periods.
30     */
31    static boolean nameIsValid(String name) {
32      return VALID_NAME.matcher(name).matches();
33    }
34  
35    /**
36     * Validate a String tag.
37     * 
38     * @param tag The String tag to validate.
39     * @return true, if the tag is 1 to 200 characters long and starts with a lowercase ASCII letter,
40     *         and contains only lowercase ASCII alphanumerics, underscores, minuses, colons, periods
41     *         and slashes.
42     */
43    static boolean tagIsValid(String tag) {
44      return VALID_TAG.matcher(tag).matches();
45    }
46  
47    /**
48     * Validate a String value.
49     * 
50     * @param value The value to validate.
51     * @return true, if value is not empty and does not have any pipe ('|') or control characters
52     */
53    static boolean fieldIsValid(String value) {
54      return NO_PIPES.matcher(value).matches();
55    }
